class Circle
{
   private : 
      //
      int x ;
      int y ;
      int r ;
   //
   public :
      //
      Circle( int xc=0 , int yc=0 ) : x(xc) , y(yc) , r(40)
      {
      }
      //
      Circle( const Circle & c )
      {
         cout << "copy" << endl ;
         //
         x = c . x ;
         y = c . y ;
         r = c . r ;
      }
      //
      ~Circle()
      {
         cout << "destructor" << endl ;
      }
      //
      bool contains( int xp , int yp ) const
      {
         int dx = xp - x ;
         int dy = yp - y ;
         //
         return dx*dx + dy*dy <= r*r ;
      }
   //
} ;
